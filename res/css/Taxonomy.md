CSS Taxonomy



font-size:
    .fs-X
    x = pixel   | max 200


Padding:
    .p-y-x
    y = top, bottom, left, right
    x = pixel   | max 250    

Margin:
    .m-y-x
    y = top, bottom, left, right
    x = pixel, auto   | max 250    

Text:
    .text-y
    y = left, right, center, right, middle


Line height:
    .lh-x-x
    x >= 1
    x <= 3

Display:
    .dis-y
    y = none, block, inline, inline-block, flex

 Position:
    .pos-y
    y = relative, absolute, fixed

 Float:
    .float-y
    y = l, r

Height and Width:
    full size:
        .sizefull

    full:
        .y-full

    maximum:
        max-y-full

    minimum:
        min-y-full

    y = height, width


Top Botton:
    y-0
    y-auto

    y = top, bottom, left, right

Opacity:
    op-x-x  |    Min 0  Max 1
